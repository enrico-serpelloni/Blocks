        function set.IdleTimeout(aPool, timeoutInMinutes)
 229 
            errorIfPoolNotUsable(aPool);
 230 
            % IdleTimeout has to be a scalar integer greater than zero
 231 
            if ~(timeoutInMinutes > 0 && timeoutInMinutes == round(timeoutInMinutes) && isscalar(timeoutInMinutes))
 232 
                error(message('parallel:lang:pool:InvalidIdleTimeout'));
 233 
            end
 234 
            
 235 
            % Setting the timeout to an effective infinite value if the
 236 
            % value request is greater than the maximum 32-bit integer (~4086 years).
 237 
            % This is to avoid accuracy issues in comparing extremely large
 238 
            % floating-point numbers. Also, the message catalog only works
 239 
            % with 32-bit integers.
 240 
            if ~isfinite(timeoutInMinutes) || timeoutInMinutes > intmax('int32')
 241 
                timeoutInSeconds = intmax('int64');
 242 
            else
 243 
                timeoutInSeconds = timeoutInMinutes * 60;
 244 
            end
 245 
            % Set the time in seconds
 246 
            aPool.hGetSession().setShutdownIdleTimeout(timeoutInSeconds);          
 247 
        end

function idleTimeoutItem = iGetIdleTimeoutDisplayableItem(diFactory, aPool)
 594 
idleTimeout = aPool.IdleTimeout;
 595 
switch idleTimeout
 596 
    case inf
 597 
        % Need to return here to avoid any time-to-timeout message
 598 
        idleTimeoutMessage = message('parallel:lang:pool:IdleTimeoutInfinite');
 599 
        idleTimeoutItem = diFactory.createDefaultItem(idleTimeoutMessage.getString());
 600 
        return;
 601 
    case 1
 602 
        idleTimeoutMessage = message('parallel:lang:pool:IdleTimeoutOneMinute');
 603 

 604 
    otherwise
 605 
        idleTimeoutMessage = message('parallel:lang:pool:IdleTimeoutFiniteManyMinutes', idleTimeout);
 606 
end
 607 

 608 
sessionInfo = aPool.hGetSession().getClientSessionInfo();
 609 
timeToIdleTimeout = ceil(sessionInfo.getRemainingSecondsBeforeShutdown()/60);
 610 

 611 
if timeToIdleTimeout <= 0
 612 
    idleTimeoutItem = diFactory.createDefaultItem(idleTimeoutMessage.getString());
 613 
    return;
 614 
end
 615 

 616 

 617 
if timeToIdleTimeout == 1
 618 
    timeToIdleTimeoutMessage = message('parallel:lang:pool:TimeToTimeoutOneMinute',...
 619 
        idleTimeout);
 620 
else
 621 
    timeToIdleTimeoutMessage = message('parallel:lang:pool:TimeToTimeoutManyMinutes',...
 622 
        idleTimeout, timeToIdleTimeout);
 623 
end
 624 
idleTimeoutItem = diFactory.createDefaultItem(timeToIdleTimeoutMessage.getString());
 625 
end
